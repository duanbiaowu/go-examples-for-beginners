# 概述

调用 `crypto/sha256` 包即可。

# 例子

```go
package main

import (
	"crypto/sha256"
	"fmt"
)

func main() {
	s := "hello world"
	h := sha256.New()
	h.Write([]byte(s))
	res := h.Sum(nil)
	fmt.Printf("sha256(`hello world`) = %x\n", res)
}

// $ go run main.go
// 输出如下
/**
  sha256(`hello world`) = b94d27b9934d3e08a52e52d7da7dabfac484efe37a5380ee9088f7ace2efcde9
*/
```
